Webcam Test Page

Demonstrates double cameras with double preview

    <script src="require.min.js" data-main="main"></script>
    <script>

  function init(rxcam) {
    const options = {
      ctx: document.querySelector('#my_camera_0'),
      config: {
        width: 400,
        height: 300,
      },
    }

    rxcam.init(options)
      .then(cam => {
        return cam.connect(0)
      })
      .catch(console.error)

    options.ctx = document.querySelector('#my_camera_1'),
    rxcam.init(options)
      .then(cam => {
        return cam.connect(1)
      })
      .catch(console.error)
  }


    </script>
