Webcam Test Page

Demonstrates ES Module and async/await

captured image will appear here...

    <script type="module">
    import { init } from '../dist/rxcam.esm.min.js'

    const options = {
      config: {
        ctx: document.querySelector('#my_camera'),
        width: 800,
        height: 600,
        previewWidth: 400,
        previewHeight: 300,
      },
    }

    init(options)
      .then(cam => {
        window.cam = cam
        return cam.connect()
      })
      .catch(console.error)
    </script>
    <script>

    async function takePhoto() {
      try {
        const url = await cam.snapshot()

        console.log('url len:' + url.length);
        document.querySelector('img#snap').src = url;
      }
      catch (ex) {
        console.error(ex)
      }
    }

    </script>
