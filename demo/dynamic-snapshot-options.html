Webcam Test Page

Demonstrates dynamic capture options

captured image will appear here...

    <script src="require.min.js" data-main="main"></script>
    <script>

  function init(rxcam) {
    const options = {
      ctx: document.querySelector('#my_camera'),
      config: {
        width: 400,
        height: 300,
      },
    }

    rxcam.init(options)
      .then(cam => {
        window.cam = cam
        return cam.connect(0)
      })
      .catch(console.error)
  }

  function takePhoto(size) {
    const opts = {
      width: 800,
      height: 600,
    }

    if (size === 'small') {
      opts.width = 400
      opts.height = 300
    }

    cam.snapshot(opts)
      .then(url => {
        console.log('url len:' + url.length);
        document.querySelector('img#snap').src = url;
      });
  }

    </script>
