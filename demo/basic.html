Webcam Test Page

Demonstrates 400x300 capture & display

captured image will appear here...

    <script src="require.min.js" data-main="main"></script>
    <script>

  function init() {
    const options = {
      config: {
        ctx: document.querySelector('#my_camera'),
        width: 400,
        height: 300,
      },
    }

    rxcam.init(options)
      .then(cam => {
        window.cam = cam
        return cam.connect()
      })
      .catch(console.error)
  }

  function takePhoto() {
    if (! cam) {
      return
    }

    cam.snapshot()
      .then(url => {
        console.log('url.len', url.length)
        document.querySelector('#snap').src = url
      })
      .catch(console.error)
  }


    </script>
