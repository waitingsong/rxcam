Webcam Test Page

Demonstrates Separated StreamConfig

captured image will appear here...

    <script type="module">
    import { init } from '../dist/rxcam.esm.min.js'

    const options = {
      ctx: document.querySelector('#my_camera'),
      config: {
        width: 400,
        height: 300,
      },
      defaultStreamConfig: {width: 800, height: 600}, // for not matched 
      streamConfigs: [
        {width: 3000, height: 2250, rotate: 90, matchLabels: ['Scanner']}, 
        {width: 1600, height: 1200, rotate: 0, matchLabels: ['USB Camera']},
      ],
    }

    init(options)
      .then(cam => {
        window.cam = cam
        return cam.connect()
      })
      .catch(console.error)
    </script>
    <script>

    async function takePhoto() {
      try {
        const url = await cam.snapshot()

        console.log('url len:' + url.length);
        document.querySelector('img#snap').src = url;
      }
      catch (ex) {
        console.error(ex)
      }
    }

    </script>
